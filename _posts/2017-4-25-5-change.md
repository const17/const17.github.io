```c
#include <stdio.h>

int main(){
 int x=10, y=20;
 int temp=0;
 
 printf("Original x:%d, y:%d\n",x,y);
 
 temp=x; // x=10, y=20, temp=10
 x=y; // x=20, y=20, temp=10
 y=temp; // x=20, y=10, temp=20
 
 printf("Changed x:%d, y:%d\n",x,y);
 
 return 0; 
}
```
